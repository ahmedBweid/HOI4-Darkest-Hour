country_event = {
	id = asia_reset_rename.283
	hidden = yes

	trigger = {
		NOT = { tag = JAP }
		owns_state = 283
		NOT = { has_country_flag = renamed_state_283 }
		# This state has been renamed by another country
		has_global_flag = renamed_state_283
	}

	mean_time_to_happen = {
		days = 3
	}

	immediate = {
		hidden_effect = {
			every_country = {
				limit = { has_country_flag = renamed_state_283 }
				clr_country_flag = renamed_state_283
			}
		}
	}

	option = {
		name = asia_reset_rename.283.a
		283 = { reset_state_name = yes }
		set_country_flag = renamed_state_283
		clr_global_flag = renamed_state_283
	}
}
